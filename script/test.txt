CREATE TABLE IF NOT EXISTS `xiaozhu`.`priceCalendar` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `luId` INT NOT NULL,
  `searchTime` DATETIME NOT NULL,
  `Day` DATETIME NOT NULL,
  `price` FLOAT NULL,
  `userSetBookable` TINYINT(1) NULL,
  `isCanBook` TINYINT(1) NULL,
  `priceType` VARCHAR(45) NULL,
  `roomNum` INT NULL,
  PRIMARY KEY (`id`))
ENGINE = InnoDB
